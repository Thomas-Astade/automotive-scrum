namespace mantis {

    repository repository {
        label "Mantis database";
        contains pojects;
        text md/mantis_tool.md; 
    }
    
    folder pojects {
        label "projects";
        contains bugs,features;
    }
    
    folder bugs {
        label "bugs";
        contains bugIssue;
    }
    
    folder features {
        label "features";
        contains featureIssue;
    }
    
    artefact bugIssue {
        label "issue";
        text md/mantis/issue.md;
    }
    
    artefact featureIssue {
        label "issue";
        text md/mantis/issue.md;
    }
    
    tool tool {
        label "Mantis";
         text md/mantis_tool.md; 
    }

    activity bugFrontDesk {
        label "bug front desk";
        brief "this activity is to check and preview entered bug issues";
        set bugIssue(New,Acknowledged)[if info is formally ok];
        set bugIssue(New,Feedback)[if info is missing];
    }

    activity createBugReport {
        label "create bug report";
        tool tool;
        set bugIssue(null,New);
     }
    
    activity createFeatureRequest {
        label "create feature request";
        tool tool;
        set featureIssue(null,New);
    }
    
}
