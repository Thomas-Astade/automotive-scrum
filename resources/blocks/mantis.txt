namespace mantis {

    repository repository {
        label "Mantis database";
        contains pojects;
        text md/mantis_tool.md; 
    }
    
    folder pojects {
        label "projects";
        contains bugs,features;
    }
    
    folder bugs {
        label "bugs";
        contains bugIssue;
    }
    
    folder features {
        label "features";
        contains featureIssue;
    }
    
    artefact bugIssue {
        label "issue";
        text md/mantis/issue.md;
    }
    
    artefact featureIssue {
        label "issue";
        text md/mantis/issue.md;
    }
    
    tool tool {
        label "Mantis";
        text md/mantis_tool.md; 
    }

    activity technicalPreview {
        label "technical preview";
        brief "This activity is to check and preview entered bug issues. There should be also a first effort estimation.";
        tool tool;
        set bugIssue(New,Acknowledged)[info is formally ok];
        set bugIssue(New,Feedback)[info is missing];
        set bugIssue(New,Closed)[duplicate];
    }

    activity bugEffortReview {
        label "effort review";
        brief "This activity is to decide, if the issue shall be considered to be done.";
        tool tool;
        set bugIssue(Acknowledged,Confirmed)[spend the estimated effort];
        set bugIssue(Acknowledged,Closed)[to much effort];
    }

    activity bugPlanning {
        label "bugfix planning";
        brief "This activity is during team planning, when an issue is assigned to a person";    
        tool tool;
        set bugIssue(Confirmed,Assigned);
    }

    activity bugFix {
        label "bug fix";
        tool tool;
        set bugIssue(Assigned,Resolved);
    }

    activity createBugReport {
        label "create bug report";
        tool tool;
        set bugIssue(null,New);
     }
    
    activity bugTest {
        label "bug test";    
        tool tool;
        set bugIssue(Resolved,Closed)[bug is solved];
        set bugIssue(Resolved,Assigned)[bug is not solved];
    }

    activity bugFeedback {
        label "get bug feedback";    
        tool tool;
        set bugIssue(Feedback,Closed)[no additional info];
        set bugIssue(Feedback,New)[new info assigned];
    }

    activity createFeatureRequest {
        label "create feature request";
        tool tool;
        set featureIssue(null,New);
    }
    
}
